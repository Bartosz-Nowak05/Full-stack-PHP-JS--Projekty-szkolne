zapytanie2

SELECT `notatki`.`nazwa`
FROM `notatki`
WHERE `notatki`.`nazwa` LIKE '%na%';

zapytanie3

SELECT `osoby`.`imie`, `notatki`.`nazwa`, `notatki`.`priorytet`
FROM `osoby`
	, `notatki`
WHERE `notatki`.`priorytet` = '5';

FROM `osoby` LEFT JOIN `notatki`
ON `osoby`.`id` = `notatki`.`id`;
